checkout.js 
// npm install  (zero extra deps)
const NexusPay = require('./sdk/nexuspay');

// Initialize with your API key
const vp = new NexusPay('vp_live_...');

// Create a payment (10 lines!)
const payment = await vp.payments.create({
  order_id:  'ORD-001',
  amount:    49900,  // ₹499 in paise
  currency:  'INR',
  customer: {
    name:  'Arjun Sharma',
    email: 'arjun@example.com',
    phone: '+919876543210'
  },
  redirect_url: 'https://yoursite.com/thanks'
});

// Redirect customer to hosted checkout
res.redirect(payment.gateway_url);
// → QR generated ✓ Email sent ✓ Webhook fires ✓
